Recipe: Multipart file upload

Goal: Accept a file from a browser form.

Forms must use enctype="multipart/form-data" or request.files stays empty.

open System.IO
open Suave
open Suave.Filters
open Suave.Operators
open Suave.Successful
open Suave.RequestErrors

let handleUpload =
  request (fun r ->
    match r.files with
    | file :: _ ->
      let dest = Path.Combine("uploads", file.fileName)
      Directory.CreateDirectory("uploads") |> ignore
      File.Copy(file.tempFilePath, dest, true)
      OK (sprintf "Saved %s" file.fileName)
    | [] -> BAD_REQUEST "No file")

let app = choose [
  GET >=> path "/" >=> OK """
    <form action="/upload" method="POST" enctype="multipart/form-data">
      <input type="file" name="file" />
      <button type="submit">Upload</button>
    </form>"""
  POST >=> path "/upload" >=> handleUpload
]

startWebServer defaultConfig app
